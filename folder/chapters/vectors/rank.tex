\begin{align}
	\myvec{\vec{B}-\vec{A}& \vec{C}-\vec{A}}^\top 
	= \myvec{1 & 4 & -4 \\ 2 & 8 & -8}
	\\
	\xleftrightarrow[]{R_2 = R_2 - 2R_1}
	 \myvec{1 & 4 & -4 \\ 0 & 0 & 0}
\end{align}
which has rank 1.  Using 
			\eqref{eq:mat-rank-t}, 
			 we conclude that the given points are collinear.
